type tStartCountingReq  = (caller: machine, key: string);
type tStartCountingResp = (key: string, val: int);

event eStartCounting     : tStartCountingReq;
event eStartCountingResp : tStartCountingResp;

machine Counter {
    var redis   : Redis;
    var caller  : machine;
    var key     : string;
    var next_id : int;

    start state Init {
        entry (input: (redis: Redis)) {
            redis   = input.redis;
            next_id = 1;
            goto Idle;
        }
    }

    state Idle {
        on eStartCounting do (req: tStartCountingReq) {
            caller = req.caller;
            key    = req.key;
            goto Counting;
        }
    }

    state Counting {
        entry {
            send redis, eRedisIncrBy,
                (caller = this, req_id = next_id, key = key, amount = 1);
        }

        on eRedisIncrResp do (resp: tRedisIncrResp) {
            next_id = next_id + 1;
            if (resp.newval < 10) {
                goto Counting;
            } else {
                send caller, eStartCountingResp, (key = key, val = resp.newval);
                goto Idle;
            }
        }

        on eRedisErrorResp do (resp: tRedisErrorResp) {
            next_id = next_id + 1;
            goto Counting;
        }
    }
}

event eInWarmUpState;
event eInReadyState;
event eErrorHappened;
event eResetPerformed;

spec EspressoMachineModesOfOperation
observes eInWarmUpState, eInReadyState, eErrorHappened, eResetPerformed
{
  start state StartUp {
    on eInWarmUpState goto WarmUp;
  }

  state WarmUp {
    on eErrorHappened goto Error;
    on eInReadyState goto Ready;
  }

  state Ready {
    ignore eInReadyState;
    on eErrorHappened goto Error;
  }

  state Error {
    on eResetPerformed goto StartUp;
    ignore eErrorHappened;
  }
}

enum tS3Status {
    S3_OK,
    S3_NOT_FOUND,
    S3_ERROR
}

type tS3UploadReq   = (caller: machine, req_id: int, bucket: string, key: string, val: any);
type tS3DownloadReq = (caller: machine, req_id: int, bucket: string, key: string);

event eS3Upload   : tS3UploadReq;
event eS3Download : tS3DownloadReq;
event eS3Down;
event eS3Up;
event eS3UploadResp   : tS3UploadResp;
event eS3ErrorResp    : tS3ErrorResp;

machine S3
{
    var store : map[string, map[string, any]];

    start state Init {
        entry {
            goto Up;
        }
    }

    state Up {
        on eS3Down goto Down;

        on eS3Upload do (req: tS3UploadReq) {
            if (!(req.bucket in store)) {
                store[req.bucket] = default(map[string, any]);
            }
            store[req.bucket][req.key] = req.val;
            send req.caller, eS3UploadResp,
                (req_id = req.req_id, bucket = req.bucket, key = req.key, status = S3_OK);
        }
    }

    state Down {
        on eS3Up goto Up;
        on eS3Upload do (req: tS3UploadReq) { send req.caller, eS3ErrorResp, (req_id = req.req_id, status = S3_ERROR); }
    }
}
